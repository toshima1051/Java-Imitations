package ifdivide;

public class if_m {

	public static void main(String[] args) {
		// mojiが"ok"だったらtrueと表示させる。
		//当てはまる条件式を選んで
		String moji = "ok";

		if() {
			System.out.println(true);
		} else {
			System.out.println(false);
		}
	}
}
