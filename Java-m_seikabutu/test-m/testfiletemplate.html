	//ある文字列に含まれる'7'の個数を計算するメソッド
	private static int countOccurrences(String strNumber, char ch) {
		int counter = 0;

		//文字分繰り返す
		for (int i = 0; i < strNumber.length(); i++) {

			//一致したならカウントする
			if (strNumber.charAt(i) == ch) {
				counter++;
			}
		}
		return counter;
	}
}
